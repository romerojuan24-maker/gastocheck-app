Interface Principal:
  ScannedDocument {
    amount: number | null;              // Monto total MXN
    date: string | null;                // YYYY-MM-DD
    vendor: string | null;              // Nombre proveedor
    concept: string | null;             // Descripción gasto
    rfc: string | null;                 // RFC formato SAT
    confidence: 'high' | 'medium' | 'low';
    warnings: string[];
  }

Función Exportada:
  scanDocument(imageBase64: string) → Promise<ScannedDocument>

Handler HTTP:
  POST /scan-document
  Body: { image_base64, mime_type? }
  Response: { ok: boolean, data?: ScannedDocument, error?: string }
